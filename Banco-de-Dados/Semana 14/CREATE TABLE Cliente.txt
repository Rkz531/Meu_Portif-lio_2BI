CREATE TABLE Cliente (
    id_cliente INTEGER PRIMARY KEY,
    nome VARCHAR(100),
    telefone VARCHAR(20),
    email VARCHAR(100)
);

CREATE TABLE Animal (
    id_animal INTEGER PRIMARY KEY,
    nome VARCHAR(100),
    especie VARCHAR(50),
    idade INTEGER,
    id_cliente INTEGER,
    FOREIGN KEY (id_cliente)
    REFERENCES Cliente(id_cliente)
);

CREATE TABLE Consulta (
    id_consulta INTEGER PRIMARY KEY,
    data_consulta DATE,
    diagnostico VARCHAR(200),
    id_animal INTEGER,
    FOREIGN KEY (id_animal) REFERENCES Animal(id_animal)
);

CREATE TABLE Vacina (
    id_vacina INTEGER PRIMARY KEY,
    nome_vacina VARCHAR(100)
);

CREATE TABLE Animal_Vacina (
    id_animal INTEGER,
    id_vacina INTEGER,
    PRIMARY KEY (id_animal, id_vacina),
    FOREIGN KEY (id_animal) REFERENCES Animal(id_animal),
    FOREIGN KEY (id_vacina) REFERENCES Vacina(id_vacina)
);

CREATE TABLE Alergia (
    id_alergia INTEGER PRIMARY KEY,
    descricao VARCHAR(100)
);

CREATE TABLE Animal_Alergia (
    id_animal INTEGER,
    id_alergia INTEGER,
    PRIMARY KEY (id_animal, id_alergia),
    FOREIGN KEY (id_animal) REFERENCES Animal(id_animal),
    FOREIGN KEY (id_alergia) REFERENCES Alergia(id_alergia)
);

Tema escolhido:
Clínica Veterinária

Problema que o sistema resolve:
Controlar clientes, animais, consultas, vacinas e alergias registradas na clínica veterinária.

Quem usaria o sistema?
Recepcionistas, veterinários e administradores da clínica.

Quais dados precisam ser armazenados?
Dados dos clientes (nome, telefone e e-mail), dados dos animais (nome, espécie e idade), consultas realizadas, vacinas aplicadas e alergias identificadas.

Principais regras de negócio:

Cada animal deve estar vinculado a um único cliente, enquanto um cliente pode possuir vários animais cadastrados. Os animais podem realizar diversas consultas ao longo do tempo, receber diferentes vacinas e possuir mais de uma alergia registrada. Todas as informações devem ser armazenadas de forma organizada, evitando duplicidade de dados e garantindo a integridade dos relacionamentos entre as tabelas.

Checklist de Normalização

Existe alguma lista dentro de uma célula?
Não.

Alguma informação aparece repetida em várias tabelas?
Não.

Dados do cliente aparecem dentro da tabela de pedidos?
Não. Os dados do cliente ficam apenas na tabela Cliente e são relacionados por chave estrangeira.

Cada tabela tem uma chave primária?
Sim.

Existe pelo menos uma chave estrangeira ligando tabelas?
Sim.
